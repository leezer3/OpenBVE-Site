The sound.cfg file format - Quick reference

Version 1.0

[Run]
Index =FileName

[Flange]
Index = FileName

[Motor]
Index = FileName

[Switch]
0 = FileName

[Brake]
BC Release High = FileName
BC Release = FileName
BC Release Full = FileName
Emergency = FileName
BP Decomp = FileName

[Compressor]
Attack = FileName
Loop = FileName
Release = FileName

[Suspension]
Left = FileName
Right = FileName

[Horn]
Primary = FileName
Secondary = FileName
Music = FileName

[Door]
Open Left = FileName
Open Right = FileName
Close Left = FileName
Close Right = FileName

[Ats]
Index = FileName

[Buzzer]
Correct = FileName

[Pilot Lamp]
On = FileName
Off = FileName

[Brake Handle]
Apply = FileName
Release = FileName
Min = FileName
Max = FileName

[Master Controller]
Up = FileName
Down = FileName
Min = FileName
Max = FileName

[Reverser]
On = FileName
Off = FileName

[Breaker]
On = FileName
Off = FileName

[Others]
Noise = FileName
Shoe = FileName
